Literate Emacs configuration. Tangled and loaded by =init.el= via
=org-babel-load-file=.

* Prelude

Ensures the tangled =config.el= starts with a =lexical-binding= cookie.

#+begin_src emacs-lisp
;;; config.el --- Tangled Emacs configuration -*- lexical-binding: t; -*-
#+end_src

* Package Management

Emacs packages are provided by Nix (see [[file:flake.nix][=flake.nix=]] and [[file:nix/hm-module.nix][=nix/hm-module.nix=]]):
the flake builds Emacs with every package below on =load-path=, so =use-package=
forms carry no =:ensure=/=:straight= and install nothing at runtime. Built-in
features (=emacs=, =eglot=, =org=, =dired=, …) use =:ensure nil=. To add a
package, add it to the =withPackages= list in =flake.nix= and rebuild.

** Garbage collection

=early-init.el= raises =gc-cons-threshold= to 1GB for fast startup but never
lowers it again, so GC almost never runs and eventually fires as a long pause
(often mid-completion). =gcmh= keeps the threshold tiny while typing and large
while idle, moving collection out of interactive work.

#+begin_src emacs-lisp
(use-package gcmh
  :init
  (gcmh-mode 1)
  :custom
  (gcmh-high-cons-threshold (* 128 1024 1024)))
#+end_src

** Themes

Auto-switches Catppuccin between Latte (light) and Mocha (dark) following the
system appearance. [[https://github.com/LionyxML/auto-dark-emacs][=auto-dark=]] provides the detection cross-platform (D-Bus on
Linux, =ns-system-appearance=/AppleScript/=osascript= on macOS), so this works
the same on Linux and Darwin, in GUI and terminal frames. When no detection
mechanism exists (e.g. a Linux TTY with no D-Bus session bus) it falls back to a
fixed dark flavor rather than warning at startup.

#+begin_src emacs-lisp
(use-package catppuccin-theme
  :config
  (defun my/catppuccin-flavor-for (appearance)
    "Return the catppuccin flavor symbol for system APPEARANCE.
APPEARANCE is `light' or `dark'.  Pure: no side effects, so it can be
unit-tested without loading a theme."
    (if (eq appearance 'light) 'latte 'mocha))
  (defun my/apply-catppuccin-flavor (appearance)
    "Set and reload the catppuccin flavor for system APPEARANCE (`light'/`dark')."
    (setq catppuccin-flavor (my/catppuccin-flavor-for appearance))
    (catppuccin-reload)
    (let ((bg (face-attribute 'mode-line :background)))
      (set-face-attribute 'mode-line nil :box `(:line-width 6 :color ,bg))
      (set-face-attribute 'mode-line-inactive nil :box `(:line-width 6 :color ,bg)))))

(use-package doom-themes
  :custom
  (doom-themes-enable-bold t)
  (doom-themes-enable-italic t)
  (doom-themes-neotree-enable-chevron-icons nil)
  :config
  (load-theme 'catppuccin t)
  (doom-themes-visual-bell-config)
  (doom-themes-neotree-config)
  (doom-themes-treemacs-config)
  (doom-themes-org-config))

(setq custom-safe-themes
      '("87fa3605a6501f9b90d337ed4d832213155e3a2e36a512984f83e847102a42f4"
        "4594d6b9753691142f02e67b8eb0fda7d12f6cc9f1299a49b819312d6addad1d"
        "7771c8496c10162220af0ca7b7e61459cb42d18c35ce272a63461c0fc1336015"
        "276228257774fa4811da55346b1e34130edb068898565ca07c2d83cfb67eb70a"
        default))

;; Drive the flavor switch off the system appearance, cross-platform. auto-dark
;; detects light/dark (D-Bus on Linux, ns-system-appearance/AppleScript on
;; macOS) and runs the mode hooks; we keep catppuccin as the single loaded theme
;; and only swap its flavor, so auto-dark must not load themes of its own
;; (hence the nil theme vars). Loaded after doom-themes so catppuccin is active
;; when `auto-dark-mode' applies the initial flavor.
;;
;; `auto-dark-allow-osascript' lets macOS detect appearance via the `osascript'
;; binary, which works in a terminal frame too (the AppleScript branch requires
;; a GUI `window-system', so `emacs -nw' would otherwise fall through).
;;
;; Some hosts have no detection mechanism at all (a Linux TTY/SSH session with
;; no D-Bus session bus). There `auto-dark-mode' would log "Could not determine
;; a viable theme detection mechanism!" at startup. `my/auto-dark-detectable-p'
;; mirrors auto-dark's own detection conditions without that warning side effect,
;; so we only enable the mode when it can work; otherwise we apply a fixed dark
;; flavor and stay quiet.
(use-package auto-dark
  :after catppuccin-theme
  :custom
  (auto-dark-dark-theme nil)
  (auto-dark-light-theme nil)
  (auto-dark-allow-osascript t)
  :config
  (defun my/auto-dark-detectable-p ()
    "Non-nil when a system-appearance detection method is available.
Mirrors `auto-dark--determine-detection-method' without its warning
side effect, so we can decide whether to enable `auto-dark-mode'."
    (cond
     ((eq system-type 'darwin)
      (or (memq window-system '(ns mac))              ; GUI -> applescript
          (and (bound-and-true-p auto-dark-allow-osascript)
               (executable-find "osascript"))))       ; terminal -> osascript
     ((eq system-type 'gnu/linux)
      (and (featurep 'dbus)
           (ignore-errors                             ; no session bus -> error
             (member "org.freedesktop.portal.Desktop"
                     (dbus-list-activatable-names :session)))))
     ((eq system-type 'windows-nt) t)
     (t nil)))
  (add-hook 'auto-dark-dark-mode-hook
            (lambda () (my/apply-catppuccin-flavor 'dark)))
  (add-hook 'auto-dark-light-mode-hook
            (lambda () (my/apply-catppuccin-flavor 'light)))
  (if (my/auto-dark-detectable-p)
      (auto-dark-mode)
    ;; No detectable mechanism (e.g. Linux TTY/SSH). Skip the mode and its
    ;; startup warning; apply a fixed dark flavor instead.
    (my/apply-catppuccin-flavor 'dark)))
#+end_src

(use-package org
  :ensure nil
  :defer t
  :config
  (setq org-startup-indented t
        org-log-done 'time
        org-hide-emphasis-markers t
        org-auto-align-tags nil
        org-tags-column 0
        org-fold-catch-invisible-edits 'show-and-error
        org-special-ctrl-a/e t
        org-insert-heading-respect-content t
        org-pretty-entities t
        org-agenda-tags-column 0
        org-ellipsis "…"
        ;; Top-level files only (non-recursive). The notes subdirectories
        ;; (0-inbox, 1-notes, 2-maps, 3-sources) hold no agenda items, so
        ;; scanning them recursively just slowed the agenda down.
        org-agenda-files '("~/notes/org/"))

  (setq +org-capture-projects-file "~/notes/org/projects.org"
        +org-capture-todo-file     "~/notes/org/todo.org"
        +org-capture-journal-file  "~/notes/org/schedule.org"
        +org-capture-notes-file    "~/notes/org/notes.org")

  ;; The (x) suffixes are fast-access keys, so C-c C-t pops up a single-keypress
  ;; selection menu instead of cycling. STARTED replaces IN-PROGRESS to match the
  ;; neovim config, whose agenda match parser can't filter hyphenated keywords.
  (setq org-todo-keywords
        '((sequence "TODO(t)" "STARTED(s)" "WAITING(w)" "|" "DONE(d)" "WONT-DO(x)")))

  (setq org-capture-templates '(
                                ("t" "Quick todo" entry
                                 (file+headline +org-capture-todo-file "Todos")
                                 "* TODO %?\n:Created: %T\n" :prepend t)
                                ("c" "Code To-Do"
                                 entry (file+headline "~/notes/org/todo.org" "Code Related Tasks")
                                 "* TODO [#B] %?\n:Created: %T\n%i\n%a\nProposed Solution: "
                                 :empty-lines 0)
                                ("e" "Event")
                                ("er" "Recurring Event" entry
                                 (file+olp +org-capture-journal-file "Events")
                                 "** EVENT %?\n%i\n" :heading "Recurring Events" :prepend nil)
                                ("eo" "One-off Event" entry
                                 (file+olp +org-capture-journal-file "Events")
                                 "** EVENT %?\n%i\n" :heading "One-off Events" :prepend nil)
                                ("w" "Work Log Entry" entry
                                 (file+datetree "~/notes/org/work-log.org")
                                 "* %?" :empty-lines 0)
                                ("n" "Note" entry
                                 (file+headline "~/notes/org/notes.org" "Random Notes")

The Roslyn server (=Microsoft.CodeAnalysis.LanguageServer=) does not auto-load a
workspace; it needs a =solution/open= (or =project/open=) notification after
init, which Eglot does not send. Without it Roslyn runs in miscellaneous-files
mode and flags every =using= directive as unused. Send it on connect, pointing
at the nearest =.sln= (top-level only) or falling back to every =.csproj=.

#+begin_src emacs-lisp
(defun my/eglot-roslyn-workspace-plan (root)
  "Return the Roslyn workspace-open plan for project ROOT, or nil.
Pure: inspects the filesystem only and returns a plist describing the
notification to send, leaving the eglot/jsonrpc call to the caller.  A
top-level .sln yields (:method :solution/open :path SLN); otherwise any
.csproj found recursively yields (:method :project/open :paths (CSPROJ...));
an empty project yields nil."
  (let ((sln (car (directory-files root t "\\.sln\\'"))))
    (cond
     (sln (list :method :solution/open :path sln))
     ((directory-files-recursively root "\\.csproj\\'")
      (list :method :project/open
            :paths (directory-files-recursively root "\\.csproj\\'")))
     (t nil))))

(defun my/eglot-roslyn-load-solution (server)
  "Point the Roslyn server at the nearest .sln (or all .csproj files).
Microsoft.CodeAnalysis.LanguageServer does not auto-load a workspace; it needs a
`solution/open' (or `project/open') notification after init, which Eglot does
not send.  Without it Roslyn runs in miscellaneous-files mode and flags every
using directive as unused."
  (when (memq 'csharp-ts-mode (eglot--major-modes server))
    (let ((plan (my/eglot-roslyn-workspace-plan
                 (project-root (eglot--project server)))))
      (pcase (plist-get plan :method)
        (:solution/open
         (jsonrpc-notify server :solution/open
                         (list :solution
                               (eglot-path-to-uri (plist-get plan :path)))))
        (:project/open
         (jsonrpc-notify server :project/open
                         (list :projects
                               (vconcat (mapcar #'eglot-path-to-uri
                                                (plist-get plan :paths))))))))))

(add-hook 'eglot-connect-hook #'my/eglot-roslyn-load-solution)
#+end_src

* JavaScript/TypeScript

The =add-node-modules-path= package defaults to running =npm bin= to locate the
directory, but that command was removed in npm v9, so it silently does nothing
on a modern toolchain. Instead, locate the nearest =node_modules/.bin= with
=locate-dominating-file= and prepend it to a buffer-local =exec-path= / =PATH=.
This is package-manager agnostic (npm/pnpm/yarn) and spawns no subprocess, and
runs on the mode hook so the project-local =typescript-language-server= (for a
manually-started =M-x eglot=) and =eslint= (for =flymake-eslint=) are already on
=exec-path=.

#+begin_src emacs-lisp
(defun my/add-node-modules-path ()
  "Prepend the nearest node_modules/.bin to a buffer-local `exec-path'.
Replaces `add-node-modules-path', whose `npm bin' default was removed in npm 9."
  (when-let* ((root (locate-dominating-file default-directory "node_modules"))
              (bin  (expand-file-name "node_modules/.bin" root)))
    (when (file-directory-p bin)
      (setq-local exec-path (cons bin exec-path))
      (setq-local process-environment
                  (cons (concat "PATH=" bin path-separator (getenv "PATH"))
                        process-environment)))))

(dolist (hook '(tsx-ts-mode-hook typescript-ts-mode-hook js-mode-hook))
  (add-hook hook #'my/add-node-modules-path))
#+end_src

[[https://github.com/dakra/ghostel][=ghostel=]] is a terminal emulator built on the Ghostty VT engine (true colour,
