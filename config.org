#+author: Hadyn Youens

#+startup: showall

Literate Emacs configuration. Tangled and loaded by =init.el= via
=org-babel-load-file=.

* Prelude

Ensures the tangled =config.el= starts with a =lexical-binding= cookie.

#+begin_src emacs-lisp
;;; config.el --- Tangled Emacs configuration -*- lexical-binding: t; -*-
#+end_src

* Package Management

Package bootstrap happens in [[file:init.el][=init.el=]] *before* this file is loaded:
[[https://github.com/radian-software/straight.el][straight.el]] is installed, and =project=, =use-package=, and =org= are pinned
to straight-managed versions. Pinning =org= early avoids a built-in/straight
version mismatch that =org-babel-load-file= would otherwise trigger.

** Garbage collection

=early-init.el= raises =gc-cons-threshold= to 1GB for fast startup but never
lowers it again, so GC almost never runs and eventually fires as a long pause
(often mid-completion). =gcmh= keeps the threshold tiny while typing and large
while idle, moving collection out of interactive work.

#+begin_src emacs-lisp
(use-package gcmh
  :ensure t
  :straight t
  :init
  (gcmh-mode 1)
  :custom
  (gcmh-high-cons-threshold (* 128 1024 1024)))
#+end_src

(use-package neotree
  :ensure t
  :straight t
  :config
  (setq neo-window-width 50)
  (setq neo-smart-open t)
  (setq neo-theme 'icons)
  (setq neo-window-position 'right)

  (defun neotree-project-root-toggle ()
    "Toggle neotree at the project root."
    (interactive)
    (let* ((proj (project-current))
           (project-dir (if proj (project-root proj) default-directory))
           (file-name (buffer-file-name)))
      (if (neo-global--window-exists-p)
          (neotree-hide)
        (progn
          (neotree-dir project-dir)
          (when file-name
            (neotree-find file-name))))))

  (define-key neotree-mode-map (kbd "r") 'neotree-rename-node)
  (define-key neotree-mode-map (kbd "c") 'neotree-copy-node)

  (global-set-key (kbd "C-x n")
                  (lambda ()
                    (interactive)
                    (neotree-dir (read-directory-name "Directory: ")))))

* Org-mode

** Core

#+begin_src emacs-lisp
(use-package org
  :ensure nil
  :defer t
  :config
  (setq org-startup-indented t
        org-log-done 'time
        org-hide-emphasis-markers t
        org-auto-align-tags nil
        org-tags-column 0
        org-fold-catch-invisible-edits 'show-and-error
        org-special-ctrl-a/e t
        org-insert-heading-respect-content t
        org-pretty-entities t
        org-agenda-tags-column 0
        org-ellipsis "…"
        ;; Top-level files plus the inbox only (non-recursive). The notes
        ;; subdirectories (1-notes, 2-maps, 3-sources) hold no agenda items,
        ;; so scanning them recursively just slowed the agenda down.
        org-agenda-files '("~/org/" "~/org/0-inbox/"))

  (setq +org-capture-projects-file "~/org/projects.org"
        +org-capture-todo-file     "~/org/todo.org"
        +org-capture-journal-file  "~/org/schedule.org"
        +org-capture-notes-file    "~/org/notes.org")

  ;; The (x) suffixes are fast-access keys, so C-c C-t pops up a single-keypress
  ;; selection menu instead of cycling. STARTED replaces IN-PROGRESS to match the
  ;; neovim config, whose agenda match parser can't filter hyphenated keywords.
  (setq org-todo-keywords
        '((sequence "TODO(t)" "STARTED(s)" "WAITING(w)" "|" "DONE(d)" "WONT-DO(x)")))

  (setq org-capture-templates '(
                                ("t" "Quick todo" entry
                                 (file+headline +org-capture-todo-file "Todos")
                                 "* TODO %?\n:Created: %T\n" :prepend t)
                                ("c" "Code To-Do"
                                 entry (file+headline "~/org/todo.org" "Code Related Tasks")
                                 "* TODO [#B] %?\n:Created: %T\n%i\n%a\nProposed Solution: "
                                 :empty-lines 0)
                                ("e" "Event")
                                ("er" "Recurring Event" entry
                                 (file+olp +org-capture-journal-file "Events")
                                 "** EVENT %?\n%i\n" :heading "Recurring Events" :prepend nil)
                                ("eo" "One-off Event" entry
                                 (file+olp +org-capture-journal-file "Events")
                                 "** EVENT %?\n%i\n" :heading "One-off Events" :prepend nil)
                                ("w" "Work Log Entry" entry
                                 (file+datetree "~/org/work-log.org")
                                 "* %?" :empty-lines 0)
                                ("n" "Note" entry
                                 (file+headline "~/org/notes.org" "Random Notes")
                                 "** %?" :empty-lines 0)))

  (setq org-todo-keyword-faces
        '(("TODO" . (:foreground "GoldenRod" :weight bold))
          ("PLANNING" . (:foreground "DeepPink" :weight bold))
          ("STARTED" . (:foreground "Cyan" :weight bold))
          ("WAITING" . (:foreground "Red" :weight bold))
          ("VERIFYING" . (:foreground "DarkOrange" :weight bold))
          ("BLOCKED" . (:foreground "Red" :weight bold))
          ("DONE" . (:foreground "LimeGreen" :weight bold))
          ("OBE" . (:foreground "LimeGreen" :weight bold))
          ("WONT-DO" . (:foreground "LimeGreen" :weight bold))))

  (setq org-tag-faces
        '(("work"      . (:foreground "mediumPurple1" :weight bold))
          ("config"    . (:foreground "royalblue1"    :weight bold))
          ("personal"  . (:foreground "forest green"  :weight bold))))

  (defun air-org-skip-subtree-if-priority (priority)
    "Skip an agenda subtree if it has a priority of PRIORITY.

    PRIORITY may be one of the characters ?A, ?B, or ?C."
    (let ((subtree-end (save-excursion (org-end-of-subtree t)))
          (pri-value (* 1000 (- org-lowest-priority priority)))
          (pri-current (org-get-priority (thing-at-point 'line t))))
      (if (= pri-value pri-current)
          subtree-end
        nil)))

  (setq org-agenda-skip-deadline-if-done t)

  (setq org-agenda-custom-commands
        '(("c" "Current Work"
           ((tags "PRIORITY=\"A\""
                  ((org-agenda-skip-function '(org-agenda-skip-entry-if 'todo 'done))
                   (org-agenda-overriding-header "High Priority:")))
            (alltodo ""
                     ((org-agenda-skip-function '(or (air-org-skip-subtree-if-priority ?A)
                                                     (org-agenda-skip-if nil '(scheduled deadline))))
                      (org-agenda-overriding-header "Other TODOs:")))
            (agenda ""
                    ((org-agenda-span 3)
                     (org-agenda-start-day "-1d")
                     (org-agenda-overriding-header "Previous Work:"))))
           ((org-agenda-tag-filter-preset '("+work"))))

          ("S" "Standup"
           ((agenda ""
                    ((org-agenda-span 1)
                     (org-agenda-start-day "-1d")
                     (org-agenda-overriding-header "Yesterday:")))
            (todo "STARTED"
                  ((org-agenda-overriding-header "Currently Working On:")))
            (todo "WAITING"

The Roslyn server (=Microsoft.CodeAnalysis.LanguageServer=) does not auto-load a
workspace; it needs a =solution/open= (or =project/open=) notification after
init, which Eglot does not send. Without it Roslyn runs in miscellaneous-files
mode and flags every =using= directive as unused. Send it on connect, pointing
at the nearest =.sln= (top-level only) or falling back to every =.csproj=.

#+begin_src emacs-lisp
(defun my/eglot-roslyn-load-solution (server)
  "Point the Roslyn server at the nearest .sln (or all .csproj files).
Microsoft.CodeAnalysis.LanguageServer does not auto-load a workspace; it needs a
`solution/open' (or `project/open') notification after init, which Eglot does
not send.  Without it Roslyn runs in miscellaneous-files mode and flags every
using directive as unused."
  (when (memq 'csharp-ts-mode (eglot--major-modes server))
    (let* ((root (project-root (eglot--project server)))
           (sln  (car (directory-files root t "\\.sln\\'"))))
      (if sln
          (jsonrpc-notify server :solution/open
                          (list :solution (eglot-path-to-uri sln)))
        (when-let* ((projects (directory-files-recursively root "\\.csproj\\'")))
          (jsonrpc-notify server :project/open
                          (list :projects
                                (vconcat (mapcar #'eglot-path-to-uri projects)))))))))

(add-hook 'eglot-connect-hook #'my/eglot-roslyn-load-solution)

* JavaScript/TypeScript

The =add-node-modules-path= package defaults to running =npm bin= to locate the
directory, but that command was removed in npm v9, so it silently does nothing
on a modern toolchain. Instead, locate the nearest =node_modules/.bin= with
=locate-dominating-file= and prepend it to a buffer-local =exec-path= / =PATH=.
This is package-manager agnostic (npm/pnpm/yarn) and spawns no subprocess, and
runs before =lsp-deferred= so the server resolves project-local binaries.

#+begin_src emacs-lisp
(defun my/add-node-modules-path ()
  "Prepend the nearest node_modules/.bin to a buffer-local `exec-path'.
Replaces `add-node-modules-path', whose `npm bin' default was removed in npm 9."
  (when-let* ((root (locate-dominating-file default-directory "node_modules"))
              (bin  (expand-file-name "node_modules/.bin" root)))
    (when (file-directory-p bin)
      (setq-local exec-path (cons bin exec-path))
      (setq-local process-environment
                  (cons (concat "PATH=" bin path-separator (getenv "PATH"))
                        process-environment)))))

(dolist (hook '(tsx-ts-mode-hook typescript-ts-mode-hook js-mode-hook))
  (add-hook hook #'my/add-node-modules-path))
#+end_src
